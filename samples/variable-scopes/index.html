Function/Local Scope

function ScopeTest () {
  var gVar = "My Global Var (Reassigned Functionally)";
  var fVar = "My Function Var";
  let fLet = "My Function Let";
  const fConst = "My Function Const";

  console.log(`${gVar}`);
  console.log(`${fVar} ${fLet} ${fConst}`);

  if (true) {
    var gVar = "My Global Var (Reassigned Locally)";
    var lVar = "My Local Var";
    let lLet = "My Local Let";
    const lConst = "My Local Const";

    //Inside Braces     console.log(`${gVar}`);
    console.log(`${fVar} ${fLet} ${fConst}`);
    console.log(`${lVar} ${lLet} ${lConst}`);
  }

  //After {} - This is valid
  console.log(gVar);
  console.log(fVar);
  console.log(lVar);
  
  //After {} - These are invalid
  console.log(lLet);
  console.log(lConst);
}
